var plus = function (a, b, c) {
  return a + b + c;
};

Function.prototype.partial = function () {
  var args = [].slice.apply(arguments);
  var self = this;
  return function () {
    return self.apply(null, args.concat([].slice.apply(arguments)));
  };
};

var plusa = plus.partial(1);
plusa(2, 3); // 6
var plusb = plusa.partial(2);
plusb(4); // 7
var plusab = plus.partial(1, 3);
plusab(5); // 9

м–ҙл–Ө н•ЁмҲҳлҘј нҳём¶ңн•  л•Ң лҢҖл¶Җл¶„мқҳ л§Өк°ң ліҖмҲҳк°Җ н•ӯмғҒ л№„мҠ·н•ҳлӢӨл©ҙ м»Өл§Ғмқ„ мӮ¬мҡ©н•  л§Ңн•ң нӣ„ліҙлқјкі  н•  мҲҳ мһҲлӢӨ. л§Өк°ңліҖмҲҳ мқјл¶ҖлҘј м Ғмҡ©н•ҳм—¬ мғҲлЎңмҡҙ н•ЁмҲҳлҘј лҸҷм ҒмңјлЎң мғқм„ұн•ҳл©ҙ мқҙ лҸҷм Ғ мғқм„ұлҗң н•ЁмҲҳлҠ” л°ҳліөм ҒмңјлЎң мӮ¬мҡ©лҗҳлҠ” л§Өк°ңліҖмҲҳлҘј лӮҙл¶Җм ҒмңјлЎң м ҖмһҘн•ҳм—¬, л§ӨлІҲ мқёмһҗлҘј м „лӢ¬н•ҳм§Җ м•Ҡм•„лҸ„ мӣҗліён•ЁмҲҳк°Җ кё°лҢҖн•ҳлҠ” кё°лҠҘмқ„ мұ„мӣҢ лҶ“кІҢ лҗ  кІғмқҙлӢӨ.

function add(x, y) {
  if (typeof y === 'undefined') {
    // л¶Җл¶„м Ғмқё м Ғмҡ©

  return x + y;
}

function multiplyThree(x) {
  return function(y) {
    return function(z) {
      return x _ y _ z;
    }
  };
}
multiplyThree(4)(8)(2); // 64

Function.prototype.curry = function(one) {
  var origFunc = this;
  var target = origFunc.length;
  var args = [];
  function next(nextOne) {
    args = args.concat(nextOne);
    if (args.length === target) {
      return origFunc.apply(null, args);
    } else {
      return function(nextOne) { return next(nextOne) };
    }
  }
  return next(one);
}

function multiplyFour(w, x, y, z) {
  return w _ x _ y \* z;
}
multiplyFour.curry(2)(3)(4)(5); // 120

1кёү к°қмІҙ (First Object, лҳҗлҠ” 1кёү мӢңлҜј)

  • 1кёү к°қмІҙ(First class object)лһҖ лӢӨмқҢкіј к°ҷмқҖ мЎ°кұҙмқ„ л§ҢмЎұн•ҳлҠ” к°қмІҙ
    • ліҖмҲҳлӮҳ лҚ°мқҙн„° кө¬мЎ°м•Ҳм—җ лӢҙмқ„ мҲҳ мһҲлӢӨ.
    • нҢҢлқјлҜён„°лЎң м „лӢ¬ н•  мҲҳ мһҲлӢӨ.
    • л°ҳнҷҳк°’(return value)мңјлЎң мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ.
    • лҸҷм ҒмңјлЎң н”„лЎңнҚјнӢ° н• лӢ№мқҙ к°ҖлҠҘн•ҳлӢӨ.

кі м°Ё н•ЁмҲҳ (High-Order Function)

  • лһҢлӢӨ кі„мӮ°лІ•м—җм„ң л§Ңл“Өм–ҙм§„ мҡ©м–ҙлЎң м•„лһҳ мЎ°кұҙмқ„ л§ҢмЎұн•ҳлҠ” н•ЁмҲҳ
    • н•ЁмҲҳм—җ н•ЁмҲҳлҘј нҢҢлқјлҜён„°лЎң м „лӢ¬н•  мҲҳ мһҲлӢӨ.
    • н•ЁмҲҳмқҳ л°ҳнҷҳк°’мңјлЎң н•ЁмҲҳлҘј мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ.

л¶ҲліҖм„ұ (Immutablility)

  • н•ЁмҲҳнҳ• н”„лЎңк·ёлһҳл°Қм—җм„ңлҠ” лҚ°мқҙн„°к°Җ ліҖн•  мҲҳ м—ҶлҠ”лҚ°, мқҙлҘј л¶ҲліҖм„ұ лҚ°мқҙн„°лқјкі  н•ңлӢӨ. (мһҗл°”мҠӨнҒ¬лҰҪнҠёлҠ” к°ҖлҠҘн•ҳм§Җл§Ң л¶Ҳк°ҖлҠҘн•ң м–ём–ҙл“Өмқҙ мһҲлӢӨ)
  • лҚ°мқҙн„° ліҖкІҪмқҙ н•„мҡ”н•ң кІҪмҡ°, мӣҗліё лҚ°мқҙн„° кө¬мЎ°лҘј ліҖкІҪн•ҳм§Җ м•Ҡкі  к·ё лҚ°мқҙн„°лҘј ліөмӮ¬ліёмқ„ л§Ңл“Өм–ҙ к·ё мқјл¶ҖлҘј ліҖкІҪн•ҳкі , ліҖкІҪн•ң ліөмӮ¬ліёмқ„ мӮ¬мҡ©н•ҙ мһ‘м—…мқ„ м§„н–үн•ңлӢӨ.
// л¶ҲліҖмқҙ м•„лӢҢ ліҖн•ҳлҠ”(Mutatable) лҚ°мқҙн„°
